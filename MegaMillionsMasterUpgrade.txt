function leFmtDate(?string $date): string
{
    if (!$date) {
        return '';
    }
    $ts = strtotime($date);
    return ($ts === false) ? '' : date('m-d-Y', $ts);
}

function leFmtDateLong(?string $date): string
{
    if (!$date) {
        return '&#8212;';
    }
    $ts = strtotime($date);
    return ($ts === false) ? '&#8212;' : date('F j, Y', $ts);
}

function lePad2(string $value): string
{
    $value = trim((string) $value);
    if ($value === '') {
        return '';
    }
    if (ctype_digit($value) && (int) $value < 10) {
        return str_pad($value, 2, '0', STR_PAD_LEFT);
    }
    return $value;
}

function leResolveLogo(string $stateAbrev, string $gName): array

function leInitRange(int $min, int $max): array
{
    $counts        = [];
    $lastSeenIndex = [];
    for ($i = $min; $i <= $max; $i++) {
        $key                 = ($i < 10) ? '0' . $i : (string) $i;
        $counts[$key]        = 0;
        $lastSeenIndex[$key] = null;
    }
    return [$counts, $lastSeenIndex];
}

function leTopKeysByValue(array $counts, int $limit, bool $ascending = false): array
{
    $work = $counts;
    if ($ascending) {
        asort($work, SORT_NUMERIC);
    } else {
        arsort($work, SORT_NUMERIC);
    }
    return array_slice(array_keys($work), 0, $limit);
}

function leFindRepeatedFromLatest(array $latestBalls, array $previousRows): array
{
    $repeated = [];
    foreach ($previousRows as $row) {
        $prevBalls = [
            trim((string) ($row['first'] ?? '')),
            trim((string) ($row['second'] ?? '')),
            trim((string) ($row['third'] ?? '')),
            trim((string) ($row['fourth'] ?? '')),
            trim((string) ($row['fifth'] ?? '')),
        ];
        foreach ($latestBalls as $ball) {
            if ($ball !== '' && in_array($ball, $prevBalls, true) && !in_array($ball, $repeated, true)) {
                $repeated[] = $ball;
            }
        }
    }
    sort($repeated, SORT_NATURAL);
    return $repeated;
}

function leCommaList(array $items): string
{
    $items = array_values(array_filter(array_map('trim', $items), static function ($v) {
        return $v !== '';
    }));
    if (empty($items)) {
        return '&#8212;';
    }
    return implode(', ', $items);
}

function leGetPreviousOccurrenceDate(
    \Joomla\Database\DatabaseDriver $db,
    string $dbCol,
    string $gameId,
    string $drawDate,
    string $ball,
    bool $isBonus = false
): ?string {
    if ($ball === '') {
        return null;
    }
    $query = $db->getQuery(true)
        ->select('MAX(' . $db->quoteName('draw_date') . ')')
        ->from($db->quoteName($dbCol))
        ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
        ->where($db->quoteName('draw_date') . ' < ' . $db->quote($drawDate));

    if ($isBonus) {
        $query->where($db->quoteName('sixth') . ' = ' . $db->quote($ball));
    } else {
        $query->where(
            '(' .
            $db->quoteName('first') . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('second') . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('third') . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('fourth') . ' = ' . $db->quote($ball) . ' OR ' .
            $db->quoteName('fifth') . ' = ' . $db->quote($ball) .
            ')'
        );
    }
    $db->setQuery($query);
    $result = $db->loadResult();
    return $result ? (string) $result : null;
}

    \Joomla\Database\DatabaseDriver $db,

        ->select('COUNT(' . $db->quoteName('id') . ')')
        ->from($db->quoteName($dbCol))

    $db->setQuery($query);
    $count = (int) $db->loadResult();

/**
 * Normalizes a play set for the Mega Millions 5/70 main matrix.
 */
function leNormalizePlay(array $numbers, int $min = 1, int $max = 70, int $limit = 5): array
{
    $ints = [];
    $seen = [];
    foreach ($numbers as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n < $min || $n > $max || isset($seen[$n])) {
            continue;
        }
        $seen[$n] = true;
        $ints[]   = $n;
        if (count($ints) >= $limit) {
            break;
        }
    }
    sort($ints, SORT_NUMERIC);
    $out = [];
    foreach ($ints as $n) {
        $out[] = ($n < 10) ? '0' . $n : (string) $n;
    }
    return $out;
}

/**
 * Builds a play from ordered pools with a fallback, for the Mega Millions 1-70 matrix.
 */
function leBuildPlayFromPools(array $pools, array $fallbackPool, int $limit = 5): array
{
    $picked = [];
    $seen   = [];
    foreach ($pools as $pool) {
        foreach ($pool as $value) {
            $n = is_int($value) ? $value : leNumberToInt((string) $value);
            if ($n < 1 || $n > 70 || isset($seen[$n])) {
                continue;
            }
            $seen[$n] = true;
            $picked[] = $n;
            if (count($picked) >= $limit) {
                break 2;
            }
        }
    }
    if (count($picked) < $limit) {
        foreach ($fallbackPool as $value) {
            $n = is_int($value) ? $value : leNumberToInt((string) $value);
            if ($n < 1 || $n > 70 || isset($seen[$n])) {
                continue;
            }
            $seen[$n] = true;
            $picked[] = $n;
            if (count($picked) >= $limit) {
                break;
            }
        }
    }
    return leNormalizePlay($picked, 1, 70, $limit);
}

function leConsecutiveCount(array $numbers): int
{
    $sorted = [];
    foreach ($numbers as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) {
            $sorted[] = $n;
        }
    }
    sort($sorted, SORT_NUMERIC);
    $count = 0;
    for ($i = 1; $i < count($sorted); $i++) {
        if (($sorted[$i] - $sorted[$i - 1]) === 1) {
            $count++;
        }
    }
    return $count;
}

function leArithmeticPatternScore(array $numbers): int
{
    $sorted = [];
    foreach ($numbers as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) {
            $sorted[] = $n;
        }
    }
    sort($sorted, SORT_NUMERIC);
    if (count($sorted) < 4) {
        return 0;
    }
    $sameDiff = 0;
    for ($i = 2; $i < count($sorted); $i++) {
        $d1 = $sorted[$i - 1] - $sorted[$i - 2];
        $d2 = $sorted[$i] - $sorted[$i - 1];
        if ($d1 === $d2) {
            $sameDiff++;
        }
    }
    return $sameDiff;
}

/**
 * Returns structural profile metrics for a Mega Millions 5/70 play set.
 * Low/high split at <=35 (midpoint of 1-70 range).
 * Cluster density: tight <=24 (roughly 1/3 of max spread 69), moderate <=38.
 */
function leBuildCombinationProfile(array $play, array $latestMain): array
{
    $ints = [];
    foreach ($play as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) {
            $ints[] = $n;
        }
    }
    sort($ints, SORT_NUMERIC);
    $odd = 0;
    $low = 0;
    foreach ($ints as $n) {
        if (($n % 2) !== 0) {
            $odd++;
        }
        if ($n <= 35) {
            $low++;
        }
    }
    $sum         = array_sum($ints);
    $spread      = !empty($ints) ? (max($ints) - min($ints)) : 0;
    $consecutive = leConsecutiveCount($ints);
    $repeated    = 0;
    $latestLookup = [];
    foreach ($latestMain as $value) {
        $latestLookup[lePad2((string) $value)] = true;
    }
    foreach ($ints as $n) {
        $key = ($n < 10) ? '0' . $n : (string) $n;
        if (isset($latestLookup[$key])) {
            $repeated++;
        }
    }
    $density = 'Wide spread';
    if ($spread <= 24) {
        $density = 'High cluster density';
    } elseif ($spread <= 38) {
        $density = 'Moderate cluster density';
    }
    return [
        'odd_even'        => $odd . '/' . (count($ints) - $odd),
        'low_high'        => $low . '/' . (count($ints) - $low),
        'sum'             => $sum,
        'spread'          => $spread,
        'consecutive'     => $consecutive,
        'repeated'        => $repeated,
        'cluster_density' => $density,
    ];
}

/**
 * Crowd-likeness scoring for Mega Millions plays.
 * Birthday threshold <=31 is universal. Spread thresholds scaled from 1-70 range.
 */
function leCrowdRiskProfile(array $play, array $favoriteNumbers): array
{
    $ints = [];
    foreach ($play as $value) {
        $n = is_int($value) ? $value : leNumberToInt((string) $value);
        if ($n > 0) {
            $ints[] = $n;
        }
    }
    sort($ints, SORT_NUMERIC);
    $under31        = 0;
    $favHits        = 0;
    $favoriteLookup = [];
    foreach ($favoriteNumbers as $fav) {
        $favoriteLookup[(int) $fav] = true;
    }
    foreach ($ints as $n) {
        if ($n <= 31) {
            $under31++;
        }
        if (isset($favoriteLookup[$n])) {
            $favHits++;
        }
    }
    $spread        = !empty($ints) ? (max($ints) - min($ints)) : 0;
    $consecutive   = leConsecutiveCount($ints);
    $visualPattern = leArithmeticPatternScore($ints) >= 2;
    $score         = 0;
    $reasons       = [];
    if ($under31 >= 4) {
        $score    += 2;
        $reasons[] = 'birthday-heavy composition';
    } elseif ($under31 >= 3) {
        $score    += 1;
        $reasons[] = 'leaning toward under-31 numbers';
    }
    if ($consecutive >= 2) {
        $score    += 2;
        $reasons[] = 'sequence concentration';
    } elseif ($consecutive === 1) {
        $score    += 1;
        $reasons[] = 'minor sequence pattern';
    }
    if ($visualPattern) {
        $score    += 1;
        $reasons[] = 'obvious spacing pattern';
    }
    if ($spread <= 28) {
        $score    += 2;
        $reasons[] = 'tight clustering';
    } elseif ($spread <= 38) {
        $score    += 1;
        $reasons[] = 'moderate clustering';
    }
    if ($favHits >= 2) {
        $score    += 1;
        $reasons[] = 'cultural favorite overlap';
    }
    $label = 'Low crowd risk';
    if ($score >= 5) {
        $label = 'High crowd risk';
    } elseif ($score >= 3) {
        $label = 'Medium crowd risk';
    }
    $explanation = !empty($reasons)
        ? ('Signals: ' . implode(', ', $reasons) . '.')
        : 'Signals: broad spread and lower overlap with common crowd patterns.';
    return [
        'score'       => $score,
        'label'       => $label,
        'explanation' => $explanation,
        'under31'     => $under31,
    ];
}

  function loadChartJsIfNeeded(done) {
    if (window.Chart) {
      done();
      return;
    }

    var cdnUrl   = 'https://cdn.jsdelivr.net/npm/chart.js@4.4.1/dist/chart.umd.min.js';
    // SRI hash from Chart.js 4.4.1 official release. The tryLoad fallback below retries without
    // integrity if the hash check fails, ensuring charts load even if the hash is stale.
    var integrity = 'sha384-OLBgp1GsljhM2TJ+sbHjaiH9txEUvgdDTAzHv2P24donTt6/529l+9Ua0vFImLlb';

    function tryLoad(withIntegrity) {
      var script = document.createElement('script');
      script.src = cdnUrl;
      if (withIntegrity) {
        script.integrity  = integrity;
        script.crossOrigin = 'anonymous';
      }
      script.async = true;
      script.onload = function () {
        done();
      };
      script.onerror = function () {
        if (withIntegrity) {
          tryLoad(false);
        } else {
          done();
        }
      };
      document.head.appendChild(script);
    }

    tryLoad(true);
  }

  function commonBarOptions(horizontal) {

      maintainAspectRatio: false,
      indexAxis: horizontal ? 'y' : 'x',

      } : {
        x: {

  function retryRenderCharts(attempts) {
    if (!window.Chart) {
      if (attempts > 0) {
        window.setTimeout(function () {
          retryRenderCharts(attempts - 1);
        }, 300);
      }
      return;
    }
    renderCharts();
  }

  function bindTrackers() {
    var mainWrap   = document.getElementById('mainTrackedWrap');
    var bonusWrap  = document.getElementById('bonusTrackedWrap');
    var clearMain  = document.getElementById('clearMainTracked');
    var clearBonus = document.getElementById('clearBonusTracked');

    if (!mainWrap || !bonusWrap) { return; }

    function renderTracked(selector, wrap, chipClass, emptyText) {
      var inputs = document.querySelectorAll(selector);
      var items  = [];
      var i;
      for (i = 0; i < inputs.length; i++) {
        if (inputs[i].checked) {
          items.push(inputs[i].value);
        }
      }
      wrap.innerHTML = '';
      if (!items.length) {
        var empty = document.createElement('div');
        empty.className = 'skai-empty';
        empty.textContent = emptyText;
        wrap.appendChild(empty);
        return;
      }
      for (i = 0; i < items.length; i++) {
        var chip = document.createElement('span');
        chip.className = 'skai-chip ' + chipClass;
        chip.textContent = items[i];
        wrap.appendChild(chip);
      }
    }

    function bindGroup(selector, wrap, chipClass, emptyText) {
      var inputs = document.querySelectorAll(selector);
      var i;
      for (i = 0; i < inputs.length; i++) {
        (function (inp) {
          inp.addEventListener('change', function () {
            renderTracked(selector, wrap, chipClass, emptyText);
          });
        }(inputs[i]));
      }
      renderTracked(selector, wrap, chipClass, emptyText);
    }

    bindGroup('.js-track-main', mainWrap, 'skai-chip--main', 'Select numbers to create a short tracked set for comparison across this page.');
    bindGroup('.js-track-bonus', bonusWrap, 'skai-chip--bonus', 'Use tracking to keep a small working set visible while you compare modules.');

    if (clearMain) {
      clearMain.addEventListener('click', function () {
        var inputs = document.querySelectorAll('.js-track-main');
        var i;
        for (i = 0; i < inputs.length; i++) {
          inputs[i].checked = false;
        }
        renderTracked('.js-track-main', mainWrap, 'skai-chip--main', 'Select numbers to create a short tracked set for comparison across this page.');
      });
    }

    if (clearBonus) {
      clearBonus.addEventListener('click', function () {
        var inputs = document.querySelectorAll('.js-track-bonus');
        var i;
        for (i = 0; i < inputs.length; i++) {
          inputs[i].checked = false;
        }
        renderTracked('.js-track-bonus', bonusWrap, 'skai-chip--bonus', 'Use tracking to keep a small working set visible while you compare modules.');
      });
    }
  }

  function bindFilters() {
    var group = document.querySelector('[data-filter-group="main"]');
    var table = document.getElementById('skai-main-table');

    if (!group || !table) { return; }

    var buttons = group.querySelectorAll('.skai-filter');
    var rows    = table.querySelectorAll('tbody tr');

    function applyFilter(filter) {
      var i;
      for (i = 0; i < rows.length; i++) {
        var tags = rows[i].getAttribute('data-tags') || '';
        if (filter === 'all' || tags.indexOf(filter) !== -1) {
          rows[i].style.display = '';
        } else {
          rows[i].style.display = 'none';
        }
      }
      for (i = 0; i < buttons.length; i++) {
        buttons[i].classList.remove('is-active');
        if (buttons[i].getAttribute('data-filter') === filter) {
          buttons[i].classList.add('is-active');
        }
      }
    }

    var k;
    for (k = 0; k < buttons.length; k++) {
      (function (btn) {
        btn.addEventListener('click', function () {
          applyFilter(btn.getAttribute('data-filter'));
        });
      }(buttons[k]));
    }
  }

  function bindStrategySaving() {
    var status        = document.getElementById('strategySaveStatus');
    var playButtons   = document.querySelectorAll('.js-save-strategy');
    var trackerButtons = document.querySelectorAll('.js-save-from-tracker');

    function sanitizeText(value) {
      // Strip to safe chars and enforce a maximum length to prevent oversized payloads.
      return String(value || '').replace(/[^0-9,\-\sA-Za-z]/g, '').replace(/\s+/g, ' ').trim().substring(0, 200);
    }

    function getCheckedValues(selector) {
      var values = [];
      var inputs = document.querySelectorAll(selector);
      var i;
      for (i = 0; i < inputs.length; i++) {
        if (inputs[i] && inputs[i].type === 'checkbox' && inputs[i].checked) {
          values.push(inputs[i].value);
        }
      }
      return values;
    }

    var strategyStore = {
      save: function (payload) {
        var safeName = sanitizeText(payload.name);
        return {
          ok: false,
          persisted: false,
          message: 'Saved as placeholder only: "' + safeName + '". Persistent account storage is not enabled in this page file.'
        };
      }
    };

    function setStatus(message) {
      if (status) {
        status.textContent = message;
      }
      var mainHint = document.getElementById('mainStrategyHint');
      if (mainHint) {
        mainHint.textContent = message;
      }
    }

    var i;
    for (i = 0; i < playButtons.length; i++) {
      (function (btn) {
        btn.addEventListener('click', function () {
          var payload = {
            name: sanitizeText(btn.getAttribute('data-strategy') || 'Strategy'),
            main: sanitizeText(btn.getAttribute('data-main') || ''),
            bonus: sanitizeText(btn.getAttribute('data-bonus') || ''),
            source: 'play-card'
          };
          var result = strategyStore.save(payload);
          setStatus((result.ok ? '' : 'Notice: ') + result.message);
        });
      }(playButtons[i]));
    }

    var j;
    for (j = 0; j < trackerButtons.length; j++) {
      (function (btn) {
        btn.addEventListener('click', function () {
          var source = sanitizeText(btn.getAttribute('data-source') || 'tracker');
          var main   = getCheckedValues('.js-track-main');
          var bonus  = getCheckedValues('.js-track-bonus');
          var payload = {
            name:   sanitizeText('Tracker strategy (' + source + ')'),
            main:   sanitizeText(main.join(', ')),
            bonus:  sanitizeText(bonus.join(', ')),
            source: source
          };
          var result = strategyStore.save(payload);
          setStatus((result.ok ? '' : 'Notice: ') + result.message);
        });
      }(trackerButtons[j]));
    }
  }

  function initAnchors() {
    var tabs = document.querySelectorAll('.skai-tab');
    if (!tabs.length) { return; }

    var i;
    for (i = 0; i < tabs.length; i++) {
      (function (tab) {
        tab.addEventListener('click', function () {
          var j;
          for (j = 0; j < tabs.length; j++) {
            tabs[j].classList.remove('skai-tab--active');
          }
          tab.classList.add('skai-tab--active');
        });
      }(tabs[i]));
    }
  }

  function init() {
    bindTrackers();
    bindFilters();
    bindStrategySaving();
    initAnchors();
    loadChartJsIfNeeded(function () {
      retryRenderCharts(5);
    });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }

}());
